206 inline unsigned int next_pow2_portable(unsigned int n) {
207  if (n == 0) {
208  return n;
209  }
210 
211  n = n - 1;
212  n |= n >> 1;
213  n |= n >> 2;
214  n |= n >> 4;
215  n |= n >> 8;
216  n |= n >> 16;
217  return n + 1;
218 }
