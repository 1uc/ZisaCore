64template <int n, class T>
65struct StaticPowImpl {
66 ANY_DEVICE_INLINE constexpr static T value(T x) {
67 return x * StaticPowImpl<n - 1, T>::value(x);
68 }
69};

71template <class T>

74};

206inline unsigned int next_pow2_portable(unsigned int n) {
207 if (n == 0) {
208 return n;
209 }
210
211 n = n - 1;
212 n |= n >> 1;
213 n |= n >> 2;
214 n |= n >> 4;
215 n |= n >> 8;
216 n |= n >> 16;
217 return n + 1;
218}
219
220inline size_t next_pow2_portable(size_t n) {
221 if (n == 0) {
222 return n;
223 }
224
225 n = n - 1;
226 n |= n >> 1;
227 n |= n >> 2;
228 n |= n >> 4;
229 n |= n >> 8;
230 n |= n >> 16;
231 n |= n >> 32;
232 return n + 1;

238unsigned int next_pow2(unsigned int n) {
239 // See: https://jameshfisher.com/2018/03/30/round-up-power-2/
240 // https://stackoverflow.com/a/63668217
241
242 static_assert(sizeof(unsigned int) == 4,
243 "Unexpected size of `unsigned int`.");
244
245#if defined(__CUDACC__) && defined(__CUDA_ARCH__)
246 return detail::next_pow2_nvcc(n);
247#elif ZISA_HAS_BUILTIN_CLZ

249#else

251#endif
